A+B Problem

http://www.lintcode.com/en/problem/a-b-problem/
The first problem of lintcode, I used bit operation to solve the problem.

class Solution {
public:
    /*
     * @param a: The first integer
     * @param b: The second integer
     * @return: The sum of a and b
     */
    int aplusb(int a, int b) {
        // write your code here, try to do it without arithmetic operators.
        int tbit = 1; int carry = 0;
        int ret = 0;
        for ( int i = 0; i < 32; ++i, tbit <<= 1 ) {
            bool aset = a & tbit;
            bool bset = b & tbit;
            if ( aset && bset ) {
                ret |= (carry <<i);
                carry = 1;
            } else if ( (aset && !bset)|(!aset && bset) ) {
                if ( carry == 0 ) {
                    ret |= 1 << i;
                    carry = 0;
                } else {
                    carry = 1;
                }
            } else {
                ret |= (carry << i);
                carry = 0;
            }
        }
    }
};
