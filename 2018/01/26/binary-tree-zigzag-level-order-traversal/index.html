binary tree zigzag level order traversal

http://www.lintcode.com/en/problem/binary-tree-zigzag-level-order-traversal/
This problem is easy. I used to vector to visit level by level. A reverse of the vector is needed to do a zigzag way, a flag of forward visit or backward visit is also needed to place all nodes in order.
{ % codeblock [zigzag traversal] [lang: c++] %}
/**

  • Definition of TreeNode:
  • class TreeNode {
  • public:
  • int val;
  • TreeNode left, right;
  • TreeNode(int val) {
  • this->val = val;
  • this->left = this->right = NULL;
  • }
  • }
    */

class Solution {
public:
/*

 * @param root: A Tree
 * @return: A list of lists of integer include the zigzag level order traversal of its nodes' values.
 */
vector<vector<int>> zigzagLevelOrder(TreeNode * root) {
    if(!root)return vector<vector<int>>();
    vector<TreeNode*> nodes, newNodes;
    vector<vector<int>> results;
    nodes.push_back(root);
    bool forward = true;
    while(!nodes.empty()){
        newNodes.clear();
        results.push_back(vector<int>());
        vector<int>& v = results.back();
        for(int i = 0; i<nodes.size(); ++i){
            TreeNode * node = nodes[i];
            v.push_back(node->val);
            if(forward){
                if(node->left)newNodes.push_back(node->left);
                if(node->right)newNodes.push_back(node->right);
            }else{
                if(node->right)newNodes.push_back(node->right);
                if(node->left)newNodes.push_back(node->left);
            }
        }
        reverse(newNodes.begin(), newNodes.end());
        nodes.swap(newNodes);
        forward = !forward;
    }
    return results;
}

};
